Scroll Sync Logic for sidebar.js

Add this at the bottom of your existing sidebar.js file:

function enableScrollSync() {
  const original = document.getElementById("original");
  const translated = document.getElementById("translated");

  let isSyncingLeft = false;
  let isSyncingRight = false;

  original.addEventListener("scroll", () => {
    if (!isSyncingRight) {
      isSyncingLeft = true;
      translated.scrollTop = original.scrollTop;
    }
    isSyncingRight = false;
  });

  translated.addEventListener("scroll", () => {
    if (!isSyncingLeft) {
      isSyncingRight = true;
      original.scrollTop = translated.scrollTop;
    }
    isSyncingLeft = false;
  });
}

// Call after rendering text
function renderParallelText(source, target) {
  const originalEl = document.getElementById("original");
  const translatedEl = document.getElementById("translated");

  originalEl.innerHTML = "<h2 class='gradient-text'>Original</h2>";
  translatedEl.innerHTML = "<h2 class='gradient-text-accent'>Translation</h2>";

  source.forEach((line) => {
    const lineEl = document.createElement("div");
    lineEl.className = "line";
    lineEl.textContent = line;
    originalEl.appendChild(lineEl);
  });

  target.forEach((line) => {
    const lineEl = document.createElement("div");
    lineEl.className = "line";
    lineEl.textContent = line;
    translatedEl.appendChild(lineEl);
  });

  enableScrollSync(); // 👈 Add this here
}


How It Works
	•	Each column (original, translated) listens for scroll events.
	•	Flags (isSyncingLeft, isSyncingRight) prevent infinite feedback loops.
	•	Vertical scroll positions (scrollTop) are synchronized as the user scrolls either pane.

UX Tip

If line spacing feels off due to different sentence lengths, you can use:

.line {
  min-height: 1.5em;
  white-space: pre-wrap;
}

This helps maintain visual alignment.
